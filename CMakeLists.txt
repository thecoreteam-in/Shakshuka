cmake_minimum_required(VERSION 3.10)\n\nproject(Shakshuka VERSION 1.0 LANGUAGES CXX)\n\nset(CMAKE_CXX_STANDARD 17)\nset(CMAKE_CXX_STANDARD_REQUIRED True)\n\n# Find packages\nfind_package(SQLite3 REQUIRED)\nfind_package(OpenSSL REQUIRED)\nfind_package(spdlog REQUIRED)\nfind_package(Qt5 REQUIRED COMPONENTS Core Widgets)\nfind_package(Boost REQUIRED)\n\n# Include directories\ninclude_directories(\n    ${SQLite3_INCLUDE_DIRS}\n    ${OPENSSL_INCLUDE_DIR}\n    ${spdlog_INCLUDE_DIRS}\n    ${Qt5_INCLUDE_DIRS}\n    ${Boost_INCLUDE_DIRS}\n)\n\n# Set source files\nset(SOURCES\n    main.cpp\n    database.cpp\n    encryption.cpp\n    taskmanager.cpp\n    scheduler.cpp\n    trayicon.cpp\n    security.cpp\n    autosave.cpp\n)\n\n# Create executable\nadd_executable(Shakshuka ${SOURCES})\n\n# Link libraries based on platform\nif(WIN32)\n    target_link_libraries(Shakshuka PRIVATE ${SQLite3_LIBRARIES} OpenSSL::SSL OpenSSL::Crypto spdlog::spdlog Qt5::Core Qt5::Widgets Boost::Boost)\nelleif(APPLE)\n    target_link_libraries(Shakshuka PRIVATE ${SQLite3_LIBRARIES} OpenSSL::SSL OpenSSL::Crypto spdlog::spdlog Qt5::Core Qt5::Widgets Boost::Boost)\nelse()\n    target_link_libraries(Shakshuka PRIVATE ${SQLite3_LIBRARIES} OpenSSL::SSL OpenSSL::Crypto spdlog::spdlog Qt5::Core Qt5::Widgets Boost::Boost)\endif()\n
